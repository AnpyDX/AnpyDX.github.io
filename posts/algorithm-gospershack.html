#include <iostream>
using namespace std;

int main() {
    int n, k;
    cin >> n >> k;
    
    k = n - k;
    
    int cur = (1 << k) - 1;
    int limit = cur << (n - k);
    
    while (cur <= limit) {
        for (int i = 1; i <= n; i++) {
            if (!(cur & (1 << n - i)))
                cout << i << " ";
        }
        cout << endl;
        
        int lb = cur & -cur;
        int r = cur + lb;
        cur = ((r ^ cur) >> __builtin_ctz(lb) + 2) | r;
    }
}
